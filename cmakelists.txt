cmake_minimum_required(VERSION 3.26.4 FATAL_ERROR)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


project(cppconsmenu VERSION 1.0)


set(
    SOURCES
    src/excs.cpp
    src/excs.hpp
    src/menu_choice.cpp
    src/menu_choice.hpp
    src/menus.cpp
    src/menus.hpp
    src/screen.cpp
    src/screen.hpp
)

add_library(cpputils STATIC ${SOURCES})
target_include_directories(cpputils PUBLIC ${CMAKE_CURRENT_LIST_DIR}/include)
